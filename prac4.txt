#include <stdio.h>
#include <stdlib.h>

struct node {
    int data;
    struct node *next;
};

int main() {
    struct node *head = NULL, *temp, *newNode;
    int num, count = 0;

    printf("Enter integers to add to linked list (0 to stop):\n");
    while (1) {
        scanf("%d", &num);
        if (num == 0)
            break;

        newNode = (struct node *) malloc(sizeof(struct node));
        newNode->data = num;
        newNode->next = NULL;

        if (head == NULL) {
            head = newNode;
        } else {
            temp = head;
            while (temp->next != NULL)
                temp = temp->next;
            temp->next = newNode;
        }
    }

    printf("\nLinked List elements: ");
    temp = head;
    while (temp != NULL) {
        printf("%d --> ", temp->data);
        temp = temp->next;
        count++;
    }
    printf("NULL");

    printf("\nTotal no. of nodes: %d", count);

    return 0;
}
