#include <stdio.h>

int binary_search(int arr[], int n, int data);

int main()
{
    int n, key, i;
    int arr[100];

    printf("\nEnter the size of array: ");
    scanf("%d", &n);

    printf("\nEnter elements (sorted): ");
    if (n <= 0)
    {
        printf("\nInvalid size!\n");
        return 0;
    }

    for (i = 0; i < n; i++)
    {
        scanf("%d", &arr[i]);
    }

    printf("\nEnter value to search: ");
    scanf("%d", &key);

    int result = binary_search(arr, n, key);
    if (result == -1)
    {
        printf("\nInvalid input!\n");
    }
    else
    {
        printf("\nSearch result: %d\n", result);
    }

    return 0;
}

int binary_search(int arr[], int n, int data)
{
    if (arr == NULL || n <= 0)
        return -1;

    int l = 0, r = n - 1, mid = 0;

    while (l <= r)
    {
        mid = (l + r) / 2;
        printf("Middle value: %d\n", arr[mid]);

        if (arr[mid] == data)
        {
            return mid;
        }
        else if (arr[mid] < data)
        {
            l = mid + 1;
            printf("Moving start to index: %d\n", l);
        }
        else
        {
            r = mid - 1;
            printf("Moving end to index: %d\n", r);
        }
    }

    return -1;
}
