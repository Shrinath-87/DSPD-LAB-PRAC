#include <stdio.h>
#include <stdlib.h>
#define MAX 5

int Q[MAX];
int front = -1;
int rear = -1;
int i;

void insert(int item) {
    if (rear == MAX - 1)
        printf("Queue is full\n");
    else {
        if (front == -1)
            front = 0;
        rear = rear + 1;
        Q[rear] = item;
    }
}

void delete() {
    if (front == -1) {
        printf("UnderFlow");
    } else {
        printf("C deleted\n", front);
        if (front == rear) {
            front = -1;
            rear = -1;
        } else
            front++;
    }
}

void display() {
    if (front == -1)
        printf("Queue Empty");
    else {
        for (i = front; i <= rear; i++)
            printf("%d\t", Q[i]);
    }
}

int main() {
    int ch, item;
    printf("1. Insert 2. Delete 3. Display 4. Exit");
    scanf("%d", &ch);
    switch (ch) {
        case 1:
            printf("Insert");
            scanf("%d", &item);
            insert(item);
            break;
        case 2:
            printf("Delete");
            delete();
            break;
        case 3:
            printf("Display");
            display();
            break;
        case 4:
            exit(0);
            break;
        default:
            printf("Invalid choice");
    }
    return 0;
}
